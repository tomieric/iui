1.普通对话框

        require(['artDialog/dialog', 'jquery'], function(dialog, $){
           var d = dialog({
               title: '欢迎',
               content: '普通对话框'
           });
           $("#btn1").on("click", function(){
               d.show();
           });
    });

3.气泡浮层

        require(['artDialog/dialog', 'jquery'], function(dialog, $){
            var d = dialog({
                quickClose: true,// 点击空白处快速关闭
                content: '气泡浮层'
            });
            $("#btn3").on("click", function(){
                d.show($("#quickClose")[0]);
            });
        });

5.给对话框左下脚添加复选框

        require(['artDialog/dialog', 'jquery'], function(dialog, $){
        var d = dialog({
            title: "给对话框左下脚添加复选框",
            content: "给对话框左下脚添加复选框",
            ok: function(){},
            statusbar: '<label><input type="checkbox"/>不再提醒</label>'
        });
        $("#btn5").on("click", function(){
            d.show();
        });
    });

6.创建 iframe 内容

        require(['artDialog/dialog-plus', 'jquery'], function(dialog, $){
            var d;
            $("#btn6").on("click", function(){
                d= dialog({
                    title: "iframe",
                    url: "http://baidu.com",
                    width: '1024',
                    height: '600',
                    fixed: true
                });
                d.show();
            });
        });
